⚙️ Configuration Supabase

Connectez votre base de données. À faire une seule fois.

1
Créer un projet Supabase gratuit

Allez sur supabase.com → New Project → région Europe West.

2
Créer les tables (SQL Editor)

Dans Supabase → SQL Editor → copiez-collez ce script :

-- Table configuration create table if not exists exam_config ( id text primary key default 'main', ecole text, institut text, matiere text, niveau text, annee text, date_exam text, heure text, duree integer default 90, session text, bareme integer default 20, consigne text, pwd_admin text default 'admin123', pwd_candidat text default '', statut text default 'ferme', created_at timestamptz default now() ); create table if not exists candidats ( id uuid primary key default gen_random_uuid(), matricule text unique not null, nom text not null, prenom text, date_naissance text, filiere text, email text, created_at timestamptz default now() ); create table if not exists questions ( id text primary key, num integer, texte text, obligatoire boolean default true, ordre integer default 0, created_at timestamptz default now() ); create table if not exists sous_questions ( id text primary key, question_id text references questions(id) on delete cascade, texte text, correcte boolean default false, pts_v numeric default 0.5, pts_f numeric default -0.25, ordre integer default 0 ); create table if not exists copies ( id uuid primary key default gen_random_uuid(), matricule text references candidats(matricule), reponses jsonb default '{}', note_brute numeric default 0, note_sur_bareme numeric default 0, date_depot timestamptz default now(), unique(matricule) ); alter table exam_config enable row level security; alter table candidats enable row level security; alter table questions enable row level security; alter table sous_questions enable row level security; alter table copies enable row level security; create policy "public_read_config" on exam_config for select using (true); create policy "public_read_candidats" on candidats for select using (true); create policy "public_update_candidats" on candidats for update using (true) with check (true); create policy "public_read_questions" on questions for select using (true); create policy "public_read_sous_questions" on sous_questions for select using (true); create policy "public_insert_copies" on copies for insert with check (true); create policy "public_read_copies" on copies for select using (true); create policy "admin_all_config" on exam_config for all using (true); create policy "admin_all_candidats" on candidats for all using (true); create policy "admin_all_questions" on questions for all using (true); create policy "admin_all_sous_questions" on sous_questions for all using (true); create policy "admin_all_copies" on copies for all using (true);
3
Entrez vos clés Supabase

Settings → API → Project URL et Publishable key
